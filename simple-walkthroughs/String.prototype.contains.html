// We often want to know if a string contains // a substring. In older versions of // ECMAscript, we had to use the "indexOf" // method. But, that returns a -1 when not // found which, according to the implicit type // conversions in ECMAscript, means true. So, // we had to do a comparison to determine the // truthiness of a substring in a string. var s = 'There is nothing to fear but me.'; var search = { good: 'nothing', bad: 'marsha' }; console.log(s.indexOf(search.good) > -1); console.log(s.indexOf(search.bad) > -1); // As a convenience, ECMAscript 6 provides the // String.prototype.contains method to return // a Boolean value rather than an integer. console.log(s.contains(search.good)); console.log(s.contains(search.bad)); // Interestingly enough, ECMAscript 5 also has // the "search" method which uses a regular // expression to determine if a string contains // a pattern. It also returns an index and suffers // from the same truthiness problem as "indexOf". console.log(s.search(/to/) > -1); // The "contains" method will accept a regular // expression, too. console.log(s.contains(/to/)); // Why is that false? When we pass the regular // expression into "contains", it calls toString // on the regular expression which yields a // a serialized version of the regular expression // that contains the boundary delimeters. console.log(/to/.toString()); // This means that "contains" actually searches // for the substring "/to/" in the string. console.log('here /to/ there'.contains(/to/));
