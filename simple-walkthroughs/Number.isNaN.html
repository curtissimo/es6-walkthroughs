// If you've used ECMAscript for any period of // time for more than just DOM manipulation, then // you know about the numbers that are not // numbers: NaN values. In previous versions of // ECMAscript, we tested those with the function // "isNaN" defined on the global execution object. // ECMAscript 6 has moved it to a static method on // the Number object. console.log(isNaN(NaN), Number.isNaN(NaN)); console.log(isNaN(5), Number.isNaN(5)); // It still works the same way with non-numerical // values, too. var date = new Date(); console.log(isNaN(null), Number.isNaN(null)); console.log(isNaN(date), Number.isNaN(date)); console.log(isNaN(false), Number.isNaN(false)); // What changes, though, is that the new version // returns false for values that are strings and // objects. console.log(isNaN({}), Number.isNaN({})); console.log(isNaN('hello'), Number.isNaN('hello'));
