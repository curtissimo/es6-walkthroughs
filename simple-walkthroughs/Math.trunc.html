// Previous to ECMAscript 6, the language provided // Math.floor which rounded a number down to the // maximum integer less than or equal to the // argument. Now, ECMAscript has the Math.trunc // method that seemingly does the same thing. console.log(Math.floor(0.9), Math.trunc(0.9)); console.log(Math.floor(1.0), Math.trunc(1.0)); console.log(Math.floor(1.1), Math.trunc(1.1)); // So, the differnce comes when we get to negative // numbers. Here we see that Math.floor(-0.9) // gets the next integer down, namely -1. On the // other hand, Math.trunc just gets rid of the .9 // part which leaves -0 which is equal to 0. console.log(Math.floor(-0.9), Math.trunc(-0.9)); console.log(Math.floor(-1.0), Math.trunc(-1.0)); console.log(Math.floor(-1.1), Math.trunc(-1.1));
