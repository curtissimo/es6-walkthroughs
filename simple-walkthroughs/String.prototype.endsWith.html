// We often want to know if a string ends with // a substring. In older versions of // ECMAscript, we could use the "lastIndexOf" // method. So, to figure out if a string ended // with another string, we had to do some math. var s = 'There is nothing to fear but me.'; var search = { good: 'me.', bad: 'you.' }; function endsWith(str, search) { var strLen = str.length; var searchLen = search.length; var index = str.lastIndexOf(search); return index === strLen - searchLen; } console.log(endsWith(s, search.good)); console.log(endsWith(s, search.bad)); // A more convenient way was to use regular // expressions and the "$" terminal symbol. function endsWith(str, search) { search = new RegExp(search + '$'); return str.search(search) > -1; } console.log(endsWith(s, search.good)); console.log(endsWith(s, search.bad)); // ECMAscript 6 now offers a convenience method // to do the same thing. console.log(s.endsWith(search.good)); console.log(s.endsWith(search.bad));
