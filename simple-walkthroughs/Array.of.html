// Array.of will create an array from a list of // items, like a literal. var a1 = [ 'a', 'b', 'c', 'd' ]; var a2 = Array.of('a', 'b', 'c', 'd'); console.log(a1, a2); // That doesn't really buy us much. The cool part // is that the "of" method is bindable to any // "this" that can be constructed. Now we can // create our own array-like objects! // Define a constructable thing. function FakeArray(len) { this.constructionLength = len; return this; } // Bind the "of" method to the FakeArray FakeArray.of = Array.of; // Use the factory method to create the object var fake = FakeArray.of('a', 'b', 'c', 'd'); // We can now iterate over our fake array! for (var i = 0; i < fake.length; i += 1) { console.log(i, fake[i]); } // And, we have access to the constructor // argument. console.log(fake.constructionLength);
