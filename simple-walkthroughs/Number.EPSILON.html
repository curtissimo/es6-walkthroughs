// Because the computer has limited space to store // the representation of floating-point numbers, // eventually we get to a value so small that its // representation is larger than can be stored. console.log(1 + .2); console.log(1 + .02); console.log(1 + .002); console.log(1 + .0002); console.log(1 + .00002); console.log(1 + .000002); console.log(1 + .0000002); console.log(1 + .00000002); console.log(1 + .000000002); console.log(1 + .0000000002); console.log(1 + .00000000002); console.log(1 + .000000000002); console.log(1 + .0000000000002); console.log(1 + .00000000000002); console.log(1 + .000000000000002); console.log(1 + .0000000000000002); // And, it's at this point that we get to a // number so small that adding it to 1 does // not change the value of 1. console.log(1 + .00000000000000002); // Until ECMAscript 6, we did not have access to // that number. And, now we do and it's called // EPSILON. It's the smallest number possible that, // when added to 1, we get a value not equal to 1. console.log(1 + Number.EPSILON); // So, let's check to see if EPSILON is greater // than the number on line 20. console.log(Number.EPSILON > .00000000000000002);
