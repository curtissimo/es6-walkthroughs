// In the days before Unicode, we had 2^8 = 256 // characters and, then, 2^16 = 65,536 characters // in our character sets. We could represent each // of those characters as a number, or code, for // the specific character encoding we used in // our software. Pre-Unicode ECMAscript provides // the String.prototype.charCodeAt method to // get the integer representations of a character // in a string. var ascii = 'Hi, Curtis'; var charCode = ascii.charCodeAt(4); var char = String.fromCharCode(charCode); console.log(charCode, char); // Then, Unicode came along with its // Supplementary Multilingual Plane and its // Supplementary Ideographic Plane and more, // characters represented by more than one // character code which Unicode calls a "code // point". Now, charCodeAt doesn't work because // it expects each character to fit into a // number less than 65,536. var unicode = 'Hi, ðŸŒ '; charCode = unicode.charCodeAt(4); char = String.fromCharCode(charCode); console.log(charCode, char); // String.prototype.codePointAt exists to resolve // the values outside the Basic Multilingual Plane. var codePoint = unicode.codePointAt(4); char = String.fromCodePoint(codePoint); console.log(codePoint, char);
